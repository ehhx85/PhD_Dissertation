    \begin{equation}%
        C_{rms} =%
        \left[%
            \dfrac{1}{n-1}%
            \sum^{n}_{i=1}%
            (x_{i} - \bar{x})^{2}%
        \right]^{1/2}%
        \label{eq:contrast-rms}%
    \end{equation}%
    % ======================================================================== %
    where,%
    % ======================================================================== %
    \begin{equation*}%
        \bar{x} =%
        \dfrac{1}{n}\sum^{n}_{i=1}(x_{i})%
    \end{equation*}%
