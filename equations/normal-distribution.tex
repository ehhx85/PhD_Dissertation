    \begin{equation}%
        f(x) =%
        \dfrac{1}{\sigma\sqrt{2\pi}}%
        \exp\left[-\dfrac{(x-x_{0})^{2}}{2\sigma^{2}}\right]%
        \label{eq:normal-distribution}%
    \end{equation}%
