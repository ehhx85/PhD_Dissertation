    \begin{equation}%
        \begin{array}{ccl}%
            w(n)%
            = \dfrac{1}{2} \left[ 1 - \cos\left( \dfrac{2 \pi n}{N - 1} \right) \right]%
            = \sin^{2}\left( \dfrac{\pi n}{N - 1} \right)%
            &%
            for%
            &%
            0 \leq n \leq N-1%
        \end{array}%
        \label{eq:hann-window}%
    \end{equation}%
