        \bar{f}(x,y) =%
        \dfrac{1}{K}%
        \sum^{K}_{i=1}%
        \begin{bmatrix}%
            f_{i}(1,1) && f_{i}(1,2) && \dots && f_{i}(1,N) \\%
            f_{i}(2,1) && f_{i}(2,2) && \dots && f_{i}(2,N) \\%
            \vdots && \vdots && \ddots && \vdots \\%
            f_{i}(M,1) && f_{i}(M,2) && \dots && f_{i}(M,N) \\%
        \end{bmatrix}%
        \label{eq:image-matrix-average}%
    \end{equation}%
